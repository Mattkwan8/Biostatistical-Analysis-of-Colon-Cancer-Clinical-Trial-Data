This dashboard explores the clinical trial data for colon cancer treatments, including survival analysis, treatment group comparisons, and model predictions.

# Kaplan-Meier Survival Curves
plot(km_fit, 
     col = c("blue", "red", "green"),  
     lty = 1:3,
     xlab = "Time (days)", 
     ylab = "Survival Probability", 
     main = "Kaplan-Meier Survival Curves by Treatment Group")
legend("topright", 
       legend = c("Obs", "Lev", "Lev+5FU"),
       col = c("blue", "red", "green"),
       lty = 1:3,
       cex = 1.2,
       box.lwd = 0,
       bg = "white")

# Survival Time vs Treatment Group
ggplot(colon_clean, aes(x = rx, y = time, fill = rx)) +
  geom_boxplot() +
  labs(title = "Survival Time by Treatment Group", x = "Treatment", y = "Survival Time (days)") +
  theme(legend.position = "none")

# Cox Model Survival Curves

     main = "Survival Curves from Cox Model")
